class ToolRegistry:
    _tools: Dict[str, Type[BaseTool]] = {}
    
    @classmethod
    def register(cls, name: str):
        def decorator(tool_class):
            cls._tools[name] = tool_class
            return tool_class
        return decorator
    
    @classmethod
    def create(cls, name: str) -> BaseTool:
        return cls._tools[name]()

2. 执行流程（责任链模式）

class ToolExecutor:
    async def execute(self, tool_call: ToolCall) -> ToolResult:
        # 1. 前置中间件
        for middleware in self._before_middlewares:
            tool_call = await middleware.process(tool_call)
        
        # 2. 执行工具
        tool = ToolRegistry.create(tool_call.name)
        result = await tool.execute(tool_call.arguments)
        
        # 3. 后置中间件
        for middleware in self._after_middlewares:
            result = await middleware.process(result)
        
        return result

3. 中间件扩展

- LoggingMiddleware: 日志记录
- RetryMiddleware: 失败重试
- CacheMiddleware: 结果缓存
- MetricsMiddleware: 性能统计

class ToolMiddleware(ABC):

#### 题目9：请详细说明混合记忆服务的实现原理，包括短期记忆和长期记忆的管理策略。

**考察点**:
- 记忆管理机制理解
- Token优化策略
- 长短期记忆融合

**参考答案要点**:

```
混合记忆服务架构：

1. 短期记忆（ShortTermMemory）
   
   目的：管理当前会话上下文
   
   策略一：滑动窗口策略
   - 保留最近N轮对话
   - 自动移除最旧消息
   - 保护系统提示词
   
   策略二：摘要缓冲策略
   - 对历史对话生成摘要
   - 保留最近完整对话
   - 压缩Token消耗

   实现：
   class ShortTermMemory:
       def __init__(self, max_tokens: int = 4000):
           self._messages: List[Message] = []
           self._max_tokens = max_tokens
       
       def add_message(self, message: Message):
           self._messages.append(message)
           self._apply_strategy()
       
       def _apply_strategy(self):
           total_tokens = sum(m.token_count for m in self._messages)
           while total_tokens > self._max_tokens:
               removed = self._messages.pop(0)
               total_tokens -= removed.token_count

2. 长期记忆（向量检索）
   
   目的：跨会话知识检索
   
   实现：
   - 对话历史向量化存储
   - 基于Query检索相关历史
   - 作为上下文补充

3. 混合记忆服务

   class HybridMemoryService:
       async def build_context(
           self,
           session_id: str,
           query: str
       ) -> List[Dict]:
           context = []
           
           # 长期记忆：检索相关历史
           long_term = await self._retrieve_long_term(query, session_id)
           if long_term:
               context.append({
                   "role": "system",
                   "content": f"[相关上下文]\n{long_term}"
               })
           
           # 短期记忆：当前会话
           short_term = self._memory_manager.get(session_id)
           context.extend(short_term.get_context())
           
           return context

Token优化效果：
- 原始对话：平均8000 tokens
- 优化后：平均3000 tokens
- 成本节省：约60%
```
